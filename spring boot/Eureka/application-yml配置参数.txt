spring:
  application:
    name: simplecloud-eureka				##名字
server:
  port: 1001					##端口
eureka:
  instance:
    hostname: localhost				##主机名
  client:
    service-url:
      defaultZone: http://${eureka.instance.hostname}:${server.port}/eureka/		##注册地址
    register-with-eureka: false				##关闭eureka禁止注册自己,否则会报异常
    fetch-registry: false
